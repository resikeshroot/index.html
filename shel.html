1 Amstrong or Not #!/bin/bash echo "Enter the number" read num n=$num sum=0 while [ $num -gt 0 ] do rem=$(($num%10)) sum=$(($sum+($rem * $rem * $rem))) num=$(($num/10)) done if [ $sum -eq $n ] then echo "$n is armstrong" else echo "$n is not armstrong" fi

12 Reverse of number #!/bin/bash echo "Enter the number:" read n sd=0 rev=0 g=$n while [ $n -gt 0 ] do sd=$(($n%10)) rev=$(($rev*10+$sd)) n=$(($n/10)) done echo "Reverse of $g is:$rev"
